\section{Opening remarks}
%==============================================================================

\begin{frame}{Opening: Problem, Users, Why AI}
  \begin{block}{Problem}
    JHU students need to discover courses, respect constraints, and build feasible schedules using SIS data and evaluations---without tab-hopping and guesswork.
  \end{block}
  \vskip 0.35em
  \begin{block}{Users}
    Undergraduates planning a term: search, shortlist, compare workload, and iterate with advising-style help.
  \end{block}
  \vskip 0.35em
  \begin{alertblock}{Why AI is essential (not bolted-on)}
    \begin{itemize}\setlength\itemsep{0.2em}
      \item \textbf{Discovery:} Semantic search + match explanations over embeddings; an agent orchestrates tools instead of static forms.
      \item \textbf{Personalization:} Onboarding $\rightarrow$ structured memories; schedule-aware chat; natural-language add/drop/swap.
      \item \textbf{Reliability:} Clarification before risky writes; unified search; parallel audits with quality gates on outputs.
    \end{itemize}
  \end{alertblock}
\end{frame}

\section{What the application does}
%==============================================================================

\subsection{Course discovery}

\begin{frame}{Demo 1 --- Course discovery}
  \begin{block}{Goal}
    Open-ended query $\rightarrow$ merged ranked results from unified \texttt{searchCourses} with explicit \texttt{matchType} and readable \texttt{matchExplanation}; clarification when the request is ambiguous.
  \end{block}
  \begin{itemize}\setlength\itemsep{0.22em}
    \item Live: NL query $\rightarrow$ cards with \texttt{matchType} (\texttt{exact} / \texttt{constraint} / \texttt{semantic} / \texttt{hybrid}) and substantive explanations, including backfill when the agent forces a narrow search payload.
    \item Optional path: ambiguous intent $\rightarrow$ disambiguation UI with saved multi-turn state.
    \item When the query names a \textbf{professor} or \textbf{specific course}, parallel \textbf{RateMyProfessor + Reddit} lookups can augment internal evals; tool use stays scoped so generic browse queries stay lightweight.
  \end{itemize}
\end{frame}

\subsection{Evaluations and attribution}

\begin{frame}{Demo 2 --- Eval summaries + attribution}
  \begin{block}{Goal}
    On-demand eval summary grounded in metrics with an inspectable source table; course metrics default to \textbf{cross-term aggregation} unless the user or tool pins a specific term.
  \end{block}
  \begin{itemize}\setlength\itemsep{0.22em}
    \item Show: ``Summarize evals'' on a course card $\rightarrow$ narrative + numeric claims tied to evidence.
    \item Open \textbf{raw / source} modal: metric names, values, terms, instructors; each claim traceable to a row in the extended attribution view.
    \item If no backing eval rows: hide raw access and do not fabricate a summary.
  \end{itemize}
\end{frame}

\subsection{Profiles, memories, and history}

\begin{frame}{Demo 3 --- Onboarding, memories, history}
  \begin{block}{Goal}
    Onboarding and chat produce structured memories; completed-course history and transcript intake feed recommendations, prerequisite logic, and audits.
  \end{block}
  \begin{itemize}\setlength\itemsep{0.22em}
    \item Show: onboarding or profile $\rightarrow$ \texttt{derived\_memories} / memory list with phrasing tuned for downstream agent use.
    \item Completed courses: add/remove from history; optional \textbf{transcript} paste or upload $\rightarrow$ review list with \texttt{matched} / \texttt{ambiguous} / \texttt{unmatched} before anything is saved.
    \item Keep one short path showing how history drives \textbf{prerequisite readiness} and exclusions in planning or audit surfaces.
  \end{itemize}
\end{frame}

\subsection{Schedule audits}

\begin{frame}{Demo 4 --- Audit (parallel checks + synthesis)}
  \begin{block}{Goal}
    From the schedule context, run a \textbf{parallelized} audit that merges normalized checks into one report: workload, schedule conflicts, preference alignment for days and times, and prerequisite readiness, then apply an audit-only \textbf{quality gate} validate $\rightarrow$ one regenerate $\rightarrow$ fallback, with \textbf{partial results} when an individual check fails.
  \end{block}
  \begin{itemize}\setlength\itemsep{0.22em}
    \item Walk findings by category with severity and evidence, including \textbf{preference alignment} against stated days and times.
    \item Call out \textbf{prerequisites} and \textbf{conflicts} alongside workload so plan fit is grounded in concrete checks, not a separate opaque score.
    \item Note the async \textbf{offline evaluator} on completed audit sends for observability; be ready in Q\&A to contrast audit vs.\ non-audit output policies if asked.
  \end{itemize}
\end{frame}

\subsection{Schedules, calendar, and chat}

\begin{frame}{Demo 5 --- Schedules, calendar, and schedule-aware chat}
  \begin{block}{Goal}
    From the dashboard, open a schedule, use the \textbf{list vs.\ weekly calendar} view, optionally add \textbf{custom events} on the grid or by natural language, then use schedule-attached chat for NL add/drop/swap with stable orchestration and integration hardening.
  \end{block}
  \begin{itemize}\setlength\itemsep{0.22em}
    \item Toggle \textbf{list / calendar}; show blocks with course code, section, and local time range; mention \textbf{custom blocks} if enabled in your build.
    \item One NL mutation on the deployed app; metrics and search in schedule chat respect \textbf{allowlisted} resolved course context so the model cannot invent schedule rows.
    \item If time: walk a \textbf{clarification} loop for an ambiguous section or add, with state saved per conversation until answers resume the flow; optionally show section-aware add or swap messaging when sections collide.
  \end{itemize}
\end{frame}

\subsection{Limitations}

\begin{frame}{Closing: Limitations}
  \begin{block}{Where Atlas still falls short}
    \small
    \begin{enumerate}\setlength\itemsep{0.28em}
      \item \textbf{Partial evaluation coverage.} Checks and QA do not yet span every intent or source; stronger semantic analysis across evals, Reddit, RMP, SIS, and chat would support deeper, more comparable scoring.
      \item \textbf{Smarter agent.} We use \texttt{gpt-4o-mini} for cost and speed; larger or higher-capacity models (more context, stronger reasoning) would help long threads and tool use at higher token cost and latency.
      \item \textbf{More edge cases.} Fine-tuned flows could cover bespoke plans: weekly templates, explicit time blocks, small constraint tweaks, and advisor-specific modes beyond generic prompts.
      \item \textbf{Courses vs.\ sections.} Flows remain course-centric in places; richer section-level control (times, seats, cross-listing) would make adds, swaps, and audits match enrollment reality more closely.
    \end{enumerate}
  \end{block}
\end{frame}

\subsection{Future work and reflection}

\begin{frame}{Closing: Future work \& reflection}
  \begin{block}{Future work}

  \end{block}
  \begin{exampleblock}{Reflection}
    Team coordination across a long build window; using AI tools as development partners; trade-offs between latency and depth, autonomy and safety, and shipping proof over perfection.
  \end{exampleblock}
\end{frame}
